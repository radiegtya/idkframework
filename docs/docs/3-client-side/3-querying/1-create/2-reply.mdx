# Create Reply

import BrowserWindow from "@site/src/components/BrowserWindow";
import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";

Create a record to `Reply` table

## Body

<BrowserWindow url="/replies">
  <h4>Data Transfer Object (DTO)</h4>

```json
{
  "content": "string",
  "postId": "string"
}
```

<h4>Example</h4>

```json
{
  "content": "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Donec efficitur interdum mauris, ut laoreet dui auctor eget. Phasellus et auctor magna",
  "postId": "6c976b04-05dc-4a6e-8154-0c82ecc21b59"
}
```

</BrowserWindow>

## Request

<Tabs>
  <TabItem value="rest_api" label="REST API" default>

  ```md title="[POST]"
  /replies
  ```

  </TabItem>
  <TabItem value="js_sdk" label="Javascript SDK">

  ```javascript
  import { IDKClient } from "@idk/client";

  const idk = new IDKClient();

  const createReply = async () => {
    const reply = await idk.replies.create({
      content:
        "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Donec efficitur interdum mauris, ut laoreet dui auctor eget. Phasellus et auctor magna",
      postId:"6c976b04-05dc-4a6e-8154-0c82ecc21b59"
    });

    return reply
  }
  ```

  </TabItem>
</Tabs>
